import yfinance as yf import pandas as pd from datetime import datetime, timedelta import os # ===== 自訂參數（請修改這裡） ===== STOCK_TICKERS = ['1155.KL', '0328.KL', '1295.KL'] # 示例：Maybank, 3REN, Top Glove。添加更多如 '6869.KL' 等 START_DATE = '2025-11-01' # 開始日期 (YYYY-MM-DD) END_DATE = '2025-11-19' # 結束日期 (YYYY-MM-DD) OUTPUT_DIR = './bursa_csvs' # 輸出資料夾 # ===== 步驟 1: 獲取所有股票的資訊（公司名稱） ===== stock_info = {} for ticker in STOCK_TICKERS: try: info = yf.Ticker(ticker).info clean_code = ticker.replace('.KL', '') # 去除 .KL stock_info[clean_code] = info.get('longName', 'Unknown Stock') print(f"下載 {clean_code}: {stock_info[clean_code]}") except Exception as e: print(f"錯誤 - {ticker}: {e}") # ===== 步驟 2: 下載歷史數據 ===== data = yf.download(STOCK_TICKERS, start=START_DATE, end=END_DATE, progress=False) # data 是 MultiIndex DataFrame: columns = [('Adj Close', '1155.KL'), ('Volume', '1155.KL'), ...] # 創建輸出資料夾 os.makedirs(OUTPUT_DIR, exist_ok=True) # ===== 步驟 3: 為每個交易日生成 CSV ===== dates = data.index.strftime('%Y-%m-%d').unique() # 實際交易日 for date_str in sorted(dates): date_obj = pd.to_datetime(date_str) filename = date_obj.strftime('%Y%m%d') + '.csv' filepath = os.path.join(OUTPUT_DIR, filename) rows = [] for ticker in STOCK_TICKERS: clean_code = ticker.replace('.KL', '') stock_name = stock_info.get(clean_code, 'Unknown') try: # 當日數據 close = data['Close'][ticker].loc[date_str] if date_str in data['Close'][ticker].index else 0 volume = data['Volume'][ticker].loc[date_str] if date_str in data['Volume'][ticker].index else 0 # 前一日數據（計算 Chg） prev_date = date_obj - timedelta(days=1) prev_close = data['Close'][ticker].loc[prev_date.strftime('%Y-%m-%d')] if prev_date.strftime('%Y-%m-%d') in data['Close'][ticker].index else close chg = close - prev_close if close > 0 else 0 rows.append({ 'Code': clean_code, 'Stock': stock_name, 'Last': round(close, 3), 'Chg': round(chg, 3), 'Vol': int(volume), 'Prv Close': round(prev_close, 3) }) except Exception as e: print(f"跳過 {ticker} 在 {date_str}: {e}") continue # 寫入 CSV if rows: df = pd.DataFrame(rows) df.to_csv(filepath, index=False) print(f"生成 {filename} ({len(rows)} 隻股票)") print("完成！CSV 檔案在資料夾:", OUTPUT_DIR)
